						public class PhysicPersonBuilder {

							private PhysicPerson physicPerson;

							public PhysicPersonBuilder() {
								this.physicPerson = new PhysicPerson();
							}

							public PhysicPersonBuilder withId(Integer id) {
								this.physicPerson.setId(id);
								return this;
							}

							public PhysicPersonBuilder withEmail(Email email) {
								if (this.physicPerson.getEmails() == null)
									this.physicPerson.setEmails(new ArrayList<Email>());

								this.physicPerson.getEmails().add(email);
								return this;
							}

							public PhysicPersonBuilder withAddress(Address address) {
								if (this.physicPerson.getAddress() == null)
									this.physicPerson.setAddress(new ArrayList<Address>());

								this.physicPerson.getAddress().add(address);
								return this;
							}

							public PhysicPerson create() {
								return this.physicPerson;
							}
						}
